
name = "Chans"
author = "redskhie"
description = [[Vermilion to Route 13 ~ Chansey catcher]]


function onPathAction()

	if getRemainingPowerPoints(2, "False Swipe") >= 1 and isPokemonUsable(2) then
		if getMapName() == "Pokecenter Vermilion" then	
			moveToMap("Vermilion City")
		elseif getMapName() == "Vermilion City" then
			if not isMounted() then
				useItem("Arcanine Mount")
			elseif isMounted() then
				moveToMap("Route 11")
			else
				moveToMap("Route 11")	
			end
		elseif getMapName() == "Route 11" then
			moveToMap("Route 11 Stop House")
		elseif getMapName() == "Route 11 Stop House" then
			moveToMap("Route 12")
		elseif getMapName() == "Route 12" then
			if not isMounted() then
				useItem("Arcanine Mount")
			elseif isMounted() then
				moveToMap("Route 13")	
			end
		elseif getMapName() == "Route 13" then
			moveToGrass()
		elseif getMapName() == "Route 14" then
			if not isMounted() then
				useItem("Arcanine Mount")
			elseif isMounted() then
				moveToMap("Route 13")
			end	
		end
	elseif isNight() and getRemainingPowerPoints(2, "False Swipe") >= 1 then
		if getMapName() == "Route 13" then
			moveToCell(18, 35)
		elseif getMapName() == "Route 14" then
			moveToGrass()
		end	
	else
		
		if getMapName() == "Route 13" then
				moveToMap("Route 12")
		elseif getMapName() == "Route 12" then
			moveToMap("Route 11 Stop House")
		elseif getMapName() == "Route 11 Stop House" then
			if not isMounted() then
				useItem("Arcanine Mount")
			elseif isMounted() then
				moveToMap("Route 11")
			else
				moveToMap("Route 11")	
			end
		elseif getMapName() == "Route 11" then
			moveToMap("Vermilion City")
		elseif getMapName() == "Vermilion City" then	
			if not isMounted() then
				useItem("Arcanine Mount")
			elseif isMounted() then
				moveToMap("Pokecenter Vermilion")
			else
				moveToMap("Pokecenter Vermilion")	
			end
	    elseif getMapName() == "Pokecenter Vermilion" then
			usePokecenter()
		end
	end
end

function onBattleAction()
	if isWildBattle() and ( isOpponentShiny() or getOpponentName() == "Chansey" ) then
		if getActivePokemonNumber() == 1 then
			return sendPokemon(2) or run() or sendAnyPokemon()
		elseif ( getActivePokemonNumber() == 2 ) and ( getOpponentHealth() > 1 ) then
			return weakAttack() or sendAnyPokemon() or run()
		elseif ( getActivePokemonNumber() == 2 ) and ( getOpponentHealth() == 1 ) then

			return useItem("Pokeball") or useItem("Great Ball") or useItem("Ultra Ball") or sendUsablePokemon() or attack() or run()
        
		end
	end
	
	return run() or sendUsablePokemon() or attack()

end
