name = "Azu"
author = "redskhie"
description = [[This script will use sync and false swipe to catch Marill and Azumarill]]

function onPathAction()


	if getRemainingPowerPoints(2, "False Swipe") >= 1  then

		if getMapName() == "Pokecenter Verdanturf" then
			moveToMap("Verdanturf Town")
		elseif getMapName() == "Verdanturf Town" then
			if not isMounted() then
				useItem("Arcanine Mount")
			elseif isMounted() then
				moveToMap("Route 117")
			end
		elseif getMapName() == "Route 117" then	
			moveToWater()
		end

	else
		if getMapName() == "Route 117" then	
			moveToMap("Verdanturf Town")
		elseif getMapName() == "Verdanturf Town" then	
			if not isMounted() then
				useItem("Arcanine Mount")
			elseif isMounted() then
				moveToMap("Pokecenter Verdanturf")
			end
	    	elseif getMapName() == "Pokecenter Verdanturf" then
			usePokecenter()
		end
	end
end

function onBattleAction()
	if isWildBattle() and ( isOpponentShiny() or getOpponentName() == "Marill" or getOpponentName() == "Azurill" or getOpponentName() == "Azumarill") then
		if getActivePokemonNumber() == 1 then
			return sendPokemon(2) or run()
		elseif ( getActivePokemonNumber() == 2 ) and ( getOpponentHealth() > 1 ) then
			return weakAttack() or run()
		elseif ( getActivePokemonNumber() == 2 ) and ( getOpponentHealth() == 1 ) then

			return useItem("Pokeball") or useItem("Great Ball") or useItem("Ultra Ball") or attack() or run()
        
		end
	end
	
	return run() or sendAnyPokemon()
	
end
